# About Session

# create new session
$ tmux new -s <sesion-name>

# edit session name
ctrl + b, $

# session termination
exit

# session halt (detached)
ctrl + b, d

# list session
$ tmux ls

# restart session (attach)
$ tmux attach -t <session-number or session-name>
